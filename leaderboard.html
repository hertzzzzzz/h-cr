function renderLeaderboard(playersArray) { const sidebarContainer = document.getElementById('leaderboard-list'); const mainGridContainer = document.getElementById('main-leaderboard-grid'); DB.sortedPlayers = playersArray .filter(p => p.is_banned !== 'TRUE' && p.is_banned !== 'true') .sort((a, b) => (parseInt(b.points) || 0) - (parseInt(a.points) || 0)); // Выводим ЛОКАЛЬНЫЙ ранг (1, 2, 3...), чтобы не было дыр в сайдбаре sidebarContainer.innerHTML = DB.sortedPlayers.slice(0, 5).map((p, index) => { const countryCode = (p.country || 'WORLD').toLowerCase(); const localRank = index + 1; return `
#${localRank}${p.nickname}
`; }).join('') + `Полный Список`; sidebarContainer.classList.add('animate-list-fade'); mainGridContainer.innerHTML = ''; currentLoaded = 0; loadMorePlayers(); } function loadMorePlayers() { const grid = document.getElementById('main-leaderboard-grid'); const btnContainer = document.getElementById('load-more-container'); const nextChunk = DB.sortedPlayers.slice(currentLoaded, currentLoaded + CHUNK_SIZE); nextChunk.forEach((p, index) => { const globalIndex = currentLoaded + index; const diffValue = parseInt(p.diff) || 0; let diffHtml = diffValue > 0 ? `+${diffValue}` : diffValue < 0 ? `-${Math.abs(diffValue)}` : `-`; const countryCode = (p.country || 'WORLD').toLowerCase(); const photoSrc = p.photo || 'images/default-profile.png'; // Строгая локальная нумерация для черного квадрата карточки (1, 2, 3...) const localRank = globalIndex + 1; // Приписка с официальным местом (если отличается от локального) const globalBadge = p.global_rank && p.global_rank !== '0' && parseInt(p.global_rank) !== localRank ? `(Gl. #${p.global_rank})` : ''; const card = document.createElement('a'); card.href = `player?id=${p.player_id}`; card.className = "bg-white border border-gray-200 flex flex-col overflow-hidden card-hover relative group block opacity-0 animate-list-fade"; card.style.animationDelay = `${(index % CHUNK_SIZE) * 0.04}s`; card.innerHTML = `
#${localRank} ${diffHtml}
${p.points || 0} PTS
${p.nickname} ${globalBadge}
`; grid.appendChild(card); }); currentLoaded += nextChunk.length; if (currentLoaded >= DB.sortedPlayers.length) { btnContainer.classList.add('hidden'); btnContainer.classList.remove('block'); } else { btnContainer.classList.remove('hidden'); btnContainer.classList.add('block'); } }
